\subsection*{Chapitre 2}
% Chapitre 2
\begin{displaymath}	% Formule somme géométrique
\begin{array}{|c|c|}
\hline
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Somme géométrique}} \\
\text{Définition somme géométrique} & \sum_{j=m}^n ar^m = \left( \frac{1 - r^{n-m+1}}{1-r} \right) \\
\hline \hline % valeur accumulée
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Valeur accumulée d'une rente}} \\	
\rowcolor{lightgray} \text{au moment} & \text{formule} \\
	\hline 	\hline
\text{du dernier versement}	& k \cdot \sx{\angln j} \\
 &  \sx{\angln j} = \left( \frac{(1+j)^n - 1}{j} \right) \\
	\hline
\text{$r$ période après le dernier versement}& k \cdot s_{\angln j}(1+j)^r \\
\text{\emph{équivalent à}} & k \cdot \left( \sx{\angl{n+r} j} - \sx{\angln j} \right) \\ 
\hline \hline		
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Valeur actualisée d'une rente}} \\
	\hline % valeur actualisée
\rowcolor{lightgray} \text{au moment} & \text{formule} \\
	\hline 	\hline
\text{fin de période}	& k \cdot a_{\angln j} \\
 &  a_{\angln j} = \left( \frac{1 - (1+j)^{-n}}{j} \right) \\
	\hline
\text{début de période}& \ax**{\angln j} = \ax{\angln j} (1+j) \\
	\hline
\text{rente perpétuelle ($n \to \infty$)} & \frac{k}{j} \\
	\hline \hline
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{plusieurs paiements dans un même année}} \\
\hline
mP \sx{\angln i}^{(m)} = mP\left( \frac{(1+i)^n -1}{i^{(m)}} \right)	& mP \sx**{\angln i}^{(m)} = \left( \frac{(1+i)^n -1}{d^{(m)}} \right)\\
mP \ax{\angln i}^{(m)} = mP\left( \frac{1-(1+i)^{-n}}{i^{(m)}} \right)	& mP \ax**{\angln i}^{(m)} = \left( \frac{1-(1+i)^{-n}}{d^{(m)}} \right)\\
\hline
\end{array}
\end{displaymath}
% croissance géométrique
\begin{displaymath}
\begin{array}{|cc|}
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Croissance des versements géométrique}} \\
\hline
\text{valeur}_{t=1} = P \gamma^{-1} \ax{\angln j^*}	& \text{valeur}_{t=0} = P \ax**{\angln j^*}\\
\text{valeur}_{t=n} = P \gamma^{n-1} \sx{\angln j^*} & \text{valeur}_{t=n+1} = P \gamma^{n} \sx**{\angln j^*} \\
\multicolumn{2}{|c|}{\text{où} j^* = \frac{1+j}{\gamma} - 1} \\
\hline \hline % si fréquence versement > que croissance géométrique
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{si fréquence versements $>$ que croissance géométrique}} \\
P \ax{\angl{m} j^*} \ax**{\angln j^*}		&\text{où }  j^* = \frac{(1+j)^m}{\gamma} -1 \\
\hline \hline % modèle d'actualisation des dividendes
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Modèle d'actualisation des dividendes}} \\
\text{Prix} = \frac{D}{1 + j - \gamma} = \frac{D}{j-g} & \text{où } \gamma = 1 + g \\
\hline \hline
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Croissance arithmétique (\emph{increasing annuities})}} \\ % croissance arithmétique
(Ia)_{\angln j} = \frac{\ax**{\angln j} - nv^n}{j} & (I\ax**{})_{\angln j} = \frac{\ax**{\angln j} - nv^n}{d_j} \\
(Is)_{\angln j} = \frac{\sx**{\angln j} - n}{j} & (I\sx**{})_{\angln j} = \frac{\sx**{\angln j} - n}{d_j} \\
\hline \hline
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{décroissance arithmétique (\emph{decreasing annuities})}} \\ % décroissance arithmétique
(Da)_{\angln j} = \frac{n - \ax{\angln j}}{j} & (D\ax**{})_{\angln j} = \frac{n - \ax{\angln j}}{d_j} \\
(Ds)_{\angln j} = \frac{n(1+j)^n - \sx{\angln j}}{j} & (D\sx**{})_{\angln j} = \frac{n(1+j)^n - \sx{\angln j}}{d_j} \\
\hline	\hline
\rowcolor{Apricot} \multicolumn{2}{|c|}{\text{Rentes à paiement continu}} \\ % Rentes à paiement continu
VA =  \int_0^n h(u) \underbrace{e^{-\int_0^u \delta_s ds}}_{e^{-\delta u} \text{ si } \delta_t = \delta\ \forall\ t} du & \vline \text{Valeur}_{t=n} = \int_0^n h(u) \underbrace{e^{\int_u^n \delta_s ds}}_{e^{\delta(n-u)} \text{ si } \delta_t = \delta\ \forall\ t} du  \\
\hline
& \\
(\bar{I}\bar{a})_{\angln i} = \frac{\ax*{\angln i} - ne^{-\delta n}}{\delta}	& (\bar{I}\bar{s})_{\angln i} = \frac{\sx*{\angln i} - n}{\delta} \\
 (\bar{D}\bar{a})_{\angln i} = \frac{n - \ax*{\angln i}}{\delta} & (\bar{D}\bar{s})_{\angln i} = \frac{ne^{\delta n} - \sx*{\angln i}}{\delta} \\
 \hline
\end{array}
\end{displaymath}

\hline	\hline

\text{deuxième forme}	& P = (Fr - Cj) \ax{\angln j} + C \\
\hline	\hline
\rowcolor{GreenYellow} \multicolumn{2}{|c|}{ \text{Prix obligation entre 2 coupons}} \\
\hline	\hline
\text{avec méthode rétrospective}	& P_t = P_0(1+j)^t \\
\text{avec méthode prospective}	& P_t = (P_1 + Fr)(1+j)^{-(1-t)}\\
\text{Prix du marché}				& \text{Prix}_t = P_t - t \times Fr \\
\hline	\hline
\end{array}
\end{displaymath}

\hline
OB_t		&	OB_t = OB_{t-1} - PR_t	& OB_t = BV_t = (Fr - Cj)\ax{\angl{n-t} j} + C \\
\hline
I_t		&	I_t = OB_{t-1} j			& I_t = (Fr - Cj)(1 - v_j^{n-t+1}) + Cj \\
\hline
PR_t		& PR_t = K_t - I_t			& PR_t = (Fr - Cj) v_j^{n-t+1}\\
\hline
PR_n		&PR_n = K_t + C - I_t			& PR_n = (Fr - Cj)(1 - v_j^{n-t+1}) + C \\
\hline
\end{array}

\subsection*{Chapitre 7}
\begin{displaymath}
\begin{array}{| c | c | c |}
\hline
\rowcolor{SeaGreen} \multicolumn{3}{|c|}{ \text{Duration de Macauley}} \\
\hline	\hline
\text{Duration}			& \frac{\sum_{t=1}^n t K_t(1+s_0(t)){-t}}{\sum_{t=1}^n K_t(1+s_0(t)){-t}} & \\
\hline
\text{Convexité}		& \frac{\sum_{t=1}^n t^2 K_t(1+s_0(t)){-t}}{\sum_{t=1}^n K_t(1+s_0(t)){-t}}& \\
\hline
\text{Duration modifiée}	& - \frac{P'(i)}{P(i)} 	& \frac{D}{1 + i}\\
\hline
\text{Convexité modifiée}	& \frac{P''(i)}{P(i)}	& \frac{C + D}{(1+i)^2} \\
\end{array}
\end{displaymath}
% Duration d'une obligation à coupon-zéro
\begin{displaymath}

\hline	\hline

\hline
\text{Convexité}	& n^2 \\
\hline
\text{MD}			& \frac{n}{1+i} \\
\hline

\hline
\end{array}
\end{displaymath}
% Approximation du prix d'une obligation
\begin{displaymath}
\begin{array}{| c | c |}
\hline
\rowcolor{SeaGreen} \multicolumn{2}{|c|}{ \text{Approximation du prix dû à un changement de taux d'intérêt}} \\
\hline	\hline
\text{Approximation linéaire}	& P(i) \approx P(i_0)(1 - h \cdot MD) \\
\hline
\text{Approximation quadratique}	& P(i) \approx P(i_0)(1 - h \cdot MD + \frac{1}{2} h^2 \cdot MC)  \\
\hline
\text{Approximation Macauley}		& P(i) \approx P(i_0) \left( \frac{1 + i_0}{1 + i_0 + h} \right)^D \\
\hline
\end{array}
\end{displaymath}
